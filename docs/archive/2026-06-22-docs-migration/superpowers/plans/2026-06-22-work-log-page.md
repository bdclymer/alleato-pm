-- Owner-only freeform work-log notes for the /work-log continuity page.

create table if not exists public.work_notes (
  id           uuid primary key default gen_random_uuid(),
  body         text not null check (char_length(body) between 1 and 20000),
  pinned       boolean not null default false,
  author_email text not null,
  created_at   timestamptz not null default now(),
  updated_at   timestamptz not null default now()
);

create index if not exists work_notes_pinned_created_idx
  on public.work_notes (pinned desc, created_at desc);

-- Keep updated_at fresh on every update.
create or replace function public.work_notes_set_updated_at()
returns trigger language plpgsql as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

drop trigger if exists trg_work_notes_updated_at on public.work_notes;
create trigger trg_work_notes_updated_at
  before update on public.work_notes
  for each row execute function public.work_notes_set_updated_at();

-- RLS: only the single workspace owner may read/write. The API routes also
-- gate on OWNER_EMAIL; this policy is defense-in-depth for the anon/auth path.
alter table public.work_notes enable row level security;

drop policy if exists work_notes_owner_all on public.work_notes;
create policy work_notes_owner_all on public.work_notes

import { getApiRouteUser } from "@/lib/supabase/server";
import { GuardrailError } from "@/lib/guardrails/errors";
import { OWNER_EMAIL } from "@/lib/navigation-config";

export async function assertOwner(where: string): Promise<{ id: string; email: string }> {
  const user = await getApiRouteUser();
  if (!user?.email) {
    throw new GuardrailError({ code: "AUTH_EXPIRED", where, message: "Authentication required." });
  }
  if (user.email.toLowerCase() !== OWNER_EMAIL.toLowerCase()) {
    throw new GuardrailError({ code: "FORBIDDEN", where, message: "Owner only." });
  }
  return { id: user.id, email: user.email };
}

import { NextResponse } from "next/server";
import { assertOwner } from "@/lib/work-log/owner";
import { createServiceClient } from "@/lib/supabase/service";
import { createWorkNoteSchema } from "@/lib/validation/work-notes";
import { GuardrailError } from "@/lib/guardrails/errors";
import { toErrorResponse } from "@/lib/api-error";

export const dynamic = "force-dynamic";

const NOTE_COLUMNS = "id, body, pinned, author_email, created_at, updated_at";

export async function GET() {
  try {
    await assertOwner("GET /api/work-notes");
    const supabase = createServiceClient();
    const { data, error } = await supabase
      .from("work_notes")
      .select(NOTE_COLUMNS)
      .order("pinned", { ascending: false })
      .order("created_at", { ascending: false });
    if (error) {
      throw new GuardrailError({ code: "INTERNAL_ERROR", where: "GET /api/work-notes", message: error.message });
    }
    return NextResponse.json({ notes: data ?? [] });
  } catch (err) {
    return toErrorResponse(err);
  }
}

export async function POST(request: Request) {
  try {
    const owner = await assertOwner("POST /api/work-notes");
    const json = await request.json().catch(() => ({}));
    const input = createWorkNoteSchema.parse(json);
    const supabase = createServiceClient();
    const { data, error } = await supabase
      .from("work_notes")
      .insert({ body: input.body, pinned: input.pinned, author_email: owner.email })
      .select(NOTE_COLUMNS)
      .single();
    if (error) {
      throw new GuardrailError({ code: "INTERNAL_ERROR", where: "POST /api/work-notes", message: error.message });
    }
    return NextResponse.json({ note: data }, { status: 201 });
  } catch (err) {
    return toErrorResponse(err);
  }
}

import { NextResponse } from "next/server";
import { assertOwner } from "@/lib/work-log/owner";
import { createServiceClient } from "@/lib/supabase/service";
import { updateWorkNoteSchema } from "@/lib/validation/work-notes";
import { GuardrailError } from "@/lib/guardrails/errors";
import { toErrorResponse } from "@/lib/api-error";

export const dynamic = "force-dynamic";

const NOTE_COLUMNS = "id, body, pinned, author_email, created_at, updated_at";

export async function PATCH(request: Request, { params }: { params: Promise<{ noteId: string }> }) {
  try {
    await assertOwner("PATCH /api/work-notes/[noteId]");
    const { noteId } = await params;
    const json = await request.json().catch(() => ({}));
    const input = updateWorkNoteSchema.parse(json);
    const supabase = createServiceClient();
    const { data, error } = await supabase
      .from("work_notes")
      .update(input)
      .eq("id", noteId)
      .select(NOTE_COLUMNS)
      .single();
    if (error) {
      throw new GuardrailError({ code: "INTERNAL_ERROR", where: "PATCH /api/work-notes/[noteId]", message: error.message });
    }
    return NextResponse.json({ note: data });
  } catch (err) {
    return toErrorResponse(err);
  }
}

export async function DELETE(_request: Request, { params }: { params: Promise<{ noteId: string }> }) {
  try {
    await assertOwner("DELETE /api/work-notes/[noteId]");
    const { noteId } = await params;
    const supabase = createServiceClient();
    const { error } = await supabase.from("work_notes").delete().eq("id", noteId);
    if (error) {
      throw new GuardrailError({ code: "INTERNAL_ERROR", where: "DELETE /api/work-notes/[noteId]", message: error.message });
    }
    return new NextResponse(null, { status: 204 });
  } catch (err) {
    return toErrorResponse(err);
  }
}

// frontend/src/hooks/use-work-notes.ts
"use client";

import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { toast } from "sonner";
import { apiFetch } from "@/lib/api-client";
import type { WorkNote, CreateWorkNoteInput, UpdateWorkNoteInput } from "@/lib/validation/work-notes";

export const workNoteKeys = {
  all: ["work-notes"] as const,
  list: () => [...workNoteKeys.all, "list"] as const,
};

export function useWorkNotes() {
  const query = useQuery({
    queryKey: workNoteKeys.list(),
    queryFn: () => apiFetch<{ notes: WorkNote[] }>("/api/work-notes"),
  });
  return { notes: query.data?.notes ?? [], isLoading: query.isLoading, error: query.error as Error | null };
}

export function useCreateWorkNote() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (input: CreateWorkNoteInput) =>
      apiFetch<{ note: WorkNote }>("/api/work-notes", { method: "POST", body: JSON.stringify(input) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: workNoteKeys.list() }),
    onError: (e: Error) => toast.error(e.message),
  });
}

export function useUpdateWorkNote() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ id, ...input }: UpdateWorkNoteInput & { id: string }) =>
      apiFetch<{ note: WorkNote }>(`/api/work-notes/${id}`, { method: "PATCH", body: JSON.stringify(input) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: workNoteKeys.list() }),
    onError: (e: Error) => toast.error(e.message),
  });
}

export function useDeleteWorkNote() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) => apiFetch(`/api/work-notes/${id}`, { method: "DELETE" }),
    onSuccess: () => qc.invalidateQueries({ queryKey: workNoteKeys.list() }),
    onError: (e: Error) => toast.error(e.message),

// frontend/src/app/api/work-log/health/route.ts
import { NextResponse } from "next/server";
import { assertOwner } from "@/lib/work-log/owner";
import { createRagServiceClient, isRagDatabaseReadsEnabled } from "@/lib/supabase/service";
import { toErrorResponse } from "@/lib/api-error";

export const dynamic = "force-dynamic";

export async function GET() {
  try {
    await assertOwner("GET /api/work-log/health");

    if (!isRagDatabaseReadsEnabled()) {
      // Never silently report "all clear" when we cannot actually read the backlog.
      return NextResponse.json({
        ok: false,
        reason: "RAG_DATABASE_READS_ENABLED is off — embedding backlog is not visible.",
        pending: {},
        totalPending: 0,
        lastSyncAt: null,
      });
    }

    const rag = createRagServiceClient();
    // Pending = rows whose embedding has not completed.
    const { data, error } = await rag
      .from("rag_document_metadata")
      .select("type, embedding_status")
      .or("embedding_status.is.null,embedding_status.eq.pending,embedding_status.eq.raw_ingested");
    if (error) {
      return NextResponse.json({ ok: false, reason: error.message, pending: {}, totalPending: 0, lastSyncAt: null });
    }

    const pending: Record<string, number> = {};
    for (const row of data ?? []) {
      const key = (row as { type: string | null }).type ?? "unknown";
      pending[key] = (pending[key] ?? 0) + 1;
    }
    const totalPending = Object.values(pending).reduce((a, b) => a + b, 0);

    // Most recent embedded row = proxy for last sync activity.
    const { data: latest } = await rag
      .from("rag_document_metadata")
      .select("updated_at")
      .order("updated_at", { ascending: false })
      .limit(1)
      .maybeSingle();

    return NextResponse.json({
      ok: totalPending === 0,
      pending,
      totalPending,
      lastSyncAt: (latest as { updated_at?: string } | null)?.updated_at ?? null,
    });
  } catch (err) {
    return toErrorResponse(err);
  }
}
```

Note: confirm `rag_document_metadata` has an `updated_at` column in `frontend/src/types/rag-database.types.ts`. If it is named differently (e.g. `created_at` or `last_embedded_at`), use the exact column the type defines. If `type`/`embedding_status` enum values differ, align the `.or(...)` filter to the real status strings (the spec lists `null`, `pending`, `raw_ingested`).

// In dev the API route runs with cwd = frontend/; the repo root is its parent.
const REPO_ROOT = path.resolve(process.cwd(), "..");

interface Commit {
  sha: string;
  message: string;
  author: string;
  date: string;
  url: string | null;
}

function parseWorkingContext(raw: string) {
  const focusMatch = raw.match(/## Current focus([\s\S]*?)(?:\n## |\n---|\s*$)/);
  const focus = focusMatch ? focusMatch[1].trim() : "";
  const lastUpdated = raw.match(/\*\*Last updated:\*\*\s*(.+)/)?.[1]?.trim() ?? null;
  const lastWorkedOnBy = raw.match(/\*\*Last worked on by:\*\*\s*(.+)/)?.[1]?.trim() ?? null;
  return { focus, lastUpdated, lastWorkedOnBy, raw };
}

async function localGit(): Promise<Commit[]> {
  const { stdout } = await execFileAsync(
    "git",
    ["log", "-30", "--pretty=format:%H%x1f%s%x1f%an%x1f%aI"],
    { cwd: REPO_ROOT },
  );
  return stdout.split("\n").filter(Boolean).map((line) => {
    const [sha, message, author, date] = line.split("\x1f");
    return { sha, message, author, date, url: `https://github.com/${REPO}/commit/${sha}` };
  });
}

async function githubCommits(token: string): Promise<Commit[]> {
  const res = await fetchWithGuardrails(
    `https://api.github.com/repos/${REPO}/commits?per_page=30`,
    { headers: { Authorization: `Bearer ${token}`, "User-Agent": "alleato-work-log", Accept: "application/vnd.github+json" } },
  );
  const json = (await res.json()) as Array<{
    sha: string;
    html_url: string;
    commit: { message: string; author: { name: string; date: string } };
  }>;
  return json.map((c) => ({
    sha: c.sha,
    message: c.commit.message.split("\n")[0],
    author: c.commit.author.name,
    date: c.commit.author.date,
    url: c.html_url,
  }));
}

async function githubWorkingContext(token: string): Promise<string | null> {
  const res = await fetchWithGuardrails(
    `https://api.github.com/repos/${REPO}/contents/WORKING_CONTEXT.md`,
    { headers: { Authorization: `Bearer ${token}`, "User-Agent": "alleato-work-log", Accept: "application/vnd.github.raw" } },
  );
  if (!res.ok) return null;
  return res.text();
}

export async function GET() {
  try {
    await assertOwner("GET /api/work-log/activity");
    const token = process.env.GITHUB_TOKEN;
    const preferLocal = process.env.NODE_ENV !== "production";

    let commits: Commit[] = [];
    let workingRaw: string | null = null;
    let source: "git" | "github" = "git";

    if (preferLocal) {
      try {
        commits = await localGit();
        workingRaw = await readFile(path.join(REPO_ROOT, "WORKING_CONTEXT.md"), "utf8");
        source = "git";
      } catch {
        // fall through to GitHub if local git/fs is unavailable
      }
    }

    if (commits.length === 0 && token) {
      commits = await githubCommits(token);
      workingRaw = await githubWorkingContext(token);
      source = "github";
    }

    if (commits.length === 0 && !token) {
      return NextResponse.json({
        commits: [],
        workingContext: null,
        source,
        reason: "No local git and no GITHUB_TOKEN set — commits unavailable in this environment.",
      });
    }

    return NextResponse.json({
      commits,
      workingContext: workingRaw ? parseWorkingContext(workingRaw) : null,
      source,
    });
  } catch (err) {
    return toErrorResponse(err);
  }
}
```

// frontend/src/hooks/use-work-log.ts
"use client";

import { useQuery } from "@tanstack/react-query";
import { apiFetch } from "@/lib/api-client";

export interface WorkLogCommit { sha: string; message: string; author: string; date: string; url: string | null; }
export interface WorkingContext { focus: string; lastUpdated: string | null; lastWorkedOnBy: string | null; raw: string; }
export interface WorkLogActivity { commits: WorkLogCommit[]; workingContext: WorkingContext | null; source: "git" | "github"; reason?: string; }
export interface WorkLogHealth { ok: boolean; pending: Record<string, number>; totalPending: number; lastSyncAt: string | null; reason?: string; }

export function useWorkLogActivity() {
  const q = useQuery({ queryKey: ["work-log", "activity"], queryFn: () => apiFetch<WorkLogActivity>("/api/work-log/activity") });
  return {
    commits: q.data?.commits ?? [],
    workingContext: q.data?.workingContext ?? null,
    source: q.data?.source,
    reason: q.data?.reason,
    isLoading: q.isLoading,
    error: q.error as Error | null,
  };
}

export function useWorkLogHealth() {
  const q = useQuery({ queryKey: ["work-log", "health"], queryFn: () => apiFetch<WorkLogHealth>("/api/work-log/health"), staleTime: 30_000 });
  return {
    health: q.data ?? null,
    isLoading: q.isLoading,
    error: q.error as Error | null,
  };
}

// frontend/src/features/work-log/commit-timeline.tsx
"use client";

import { useWorkLogActivity } from "@/hooks/use-work-log";

function dayLabel(iso: string) {
  return new Date(iso).toLocaleDateString(undefined, { weekday: "short", month: "short", day: "numeric" });
}

export function CommitTimeline() {
  const { commits, isLoading, reason } = useWorkLogActivity();
  if (isLoading) return <div className="text-sm text-muted-foreground">Loading commits…</div>;
  if (commits.length === 0) {
    return <div className="text-sm text-muted-foreground">{reason ?? "No recent commits."}</div>;
  }
  const groups = new Map<string, typeof commits>();
  for (const c of commits) {
    const key = dayLabel(c.date);
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(c);
  }
  return (
    <div className="space-y-4">
      {[...groups.entries()].map(([day, items]) => (
        <div key={day} className="space-y-1.5">
          <div className="text-xs font-medium uppercase tracking-wide text-muted-foreground">{day}</div>
          {items.map((c) => (
            <div key={c.sha} className="flex items-baseline gap-3 text-sm">
              <a href={c.url ?? "#"} target="_blank" rel="noreferrer" className="font-mono text-xs text-muted-foreground hover:text-primary">
                {c.sha.slice(0, 7)}
              </a>
              <span className="flex-1 text-foreground">{c.message}</span>
              <span className="shrink-0 text-xs text-muted-foreground">{c.author}</span>
            </div>
          ))}
        </div>
      ))}
    </div>
  );
}

"use client";

import { useState } from "react";
import { useWorkNotes, useCreateWorkNote, useUpdateWorkNote, useDeleteWorkNote } from "@/hooks/use-work-notes";
import { EmptyState } from "@/components/ds";

          rows={3}
        />
        <div className="flex justify-end">
          <Button size="sm" onClick={submit} disabled={create.isPending || !draft.trim()}>Add note</Button>
        </div>
      </div>

      {isLoading ? (
        <div className="text-sm text-muted-foreground">Loading notes…</div>
      ) : notes.length === 0 ? (
        <EmptyState icon={<NotebookPen />} title="No notes yet" description="Jot down where you left off so future-you remembers." />
      ) : (
        <ul className="space-y-2">
          {notes.map((n) => (

                <div className="flex shrink-0 gap-1 opacity-0 transition group-hover:opacity-100">
                  <Button variant="ghost" size="icon" aria-label={n.pinned ? "Unpin" : "Pin"}
                    onClick={() => update.mutate({ id: n.id, pinned: !n.pinned })}>
                    {n.pinned ? <PinOff className="h-4 w-4" /> : <Pin className="h-4 w-4" />}
                  </Button>
                  <Button variant="ghost" size="icon" aria-label="Delete" onClick={() => setPendingDelete(n.id)}>
                    <Trash2 className="h-4 w-4" />
                  </Button>
                </div>
              </div>
              <div className="mt-1 text-xs text-muted-foreground">
                {n.pinned ? "Pinned · " : ""}{new Date(n.created_at).toLocaleString()}
              </div>
            </li>
          ))}
        </ul>
      )}

      <ConfirmDeleteDialog
        open={pendingDelete !== null}
        onOpenChange={(o) => !o && setPendingDelete(null)}
        onConfirm={() => { if (pendingDelete) remove.mutate(pendingDelete); setPendingDelete(null); }}

        description="This cannot be undone."

}
